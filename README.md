# coastwatch_oscar
Plot CoastWatch (NOAA) and OSCAR (JPL) original and DINEOF (NCSU) fields for the Carolinas  

conda env create -n coastwatch_oscar -f=coastwatch_oscar.yml    
conda activate coastwatch_oscar    
python coastwatch_oscar.py
